namespace = vu_bear

country_event = { # Player choice to spawn Bear Kingdom
	id = vu_bear.1
	title = vu_bear.1.t
	desc = vu_bear.1.d
	picture = LAND_MILITARY_eventPicture
	
	major = yes
	
	fire_only_once = yes
	
	goto = 5591
	
	trigger = {
		ai = no
		NOT = { exists = JMN }
		owns_or_non_sovereign_subject_of = 5591
	}
	
	mean_time_to_happen = { #About 10 years
		months = 120
	}
	
	option = { # THE BEAR HAS LANDED!
		name = vu_bear.1.a
		highlight = yes
		goto = 5591
		ai_chance = { factor = 0 }
		custom_tooltip = vu_bear.1.tooltip
		hidden_effect = {
			5591 = {
				add_core = JMN
				set_base_tax = 8
				set_base_production = 8
				set_base_manpower = 8
				add_prosperity = 100
				change_culture = jan_mayenese
				change_trade_goods = gems
				center_of_trade = 1
				if = {
					limit = {
						NOT = { fort_level = 2 }
					}
					add_building = fort_15th
				}
				if = {
					limit = {
						NOT = { has_building = workshop }
					}
					add_building = workshop
				}
				if = {
					limit = {
						NOT = { has_building = shipyard }
						NOT = { has_building = grand_shipyard }
					}
					add_building = shipyard
				}
				cede_province = ROOT
			}
			release = JMN
			5591 = {
				remove_core = ROOT
			}
			JMN = {
				build_to_forcelimit = {
					heavy_ship = 0.3
					light_ship = 0.3
					galley = 0.2
					transport = 0.2
				}
				build_to_forcelimit = {
					infantry = 0.6
					cavalry = 0.1
					artillery = 0.3
				}
				add_adm_power = 150
				add_dip_power = 150
				add_mil_power = 150
				add_sailors = 2000
				add_manpower = 10 #Bearpower rather
				add_army_tradition = 25
				add_navy_tradition = 25
				add_treasury = 1000
				add_stability = 2
				adopt_reform_progress = ROOT
			}
			set_global_flag = bear_nation_is_real
			country_event = { 
				id = vu_bear.2
				days = 1
			}
		}
	}
	
	option = { # Secret bear kingdom on Jan Mayen is not real and it's just false propaganda!
		name = vu_bear.1.b
		ai_chance = { factor = 100 }
	}
}
