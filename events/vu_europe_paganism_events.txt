namespace = vu_pagan

country_event = {	#Norsemen return from Greenland, as last colonies fall
	id = vu_pagan.1
	title = vu_pagan.1.t
	desc = vu_pagan.1.d
	picture = EUROPEAN_REFUGEES_eventPicture
	
	major = yes
	
	fire_only_once = yes
	
	goto = 3114
	
	trigger = {
		OR = {
			owns_core_province = 3114
			tag = ICE
		}
		is_year = 1450
		NOT = { is_year = 1500 }
	}
	
	mean_time_to_happen = { #About 5 years
		months = 60
	}
	
	option = { #they can settle in 3114, no repression
		name = vu_pagan.1.a
		goto = 3114
		ai_chance = { factor = 40 }
		3114 = {
			add_base_manpower = 1
			change_religion = norse_pagan_reformed
			add_local_autonomy = 30
		}
		add_prestige = -5
		add_adm_power = -20
	}
	
	option = { #They have to convert before they settle!
		name = vu_pagan.1.b
		ai_chance = { factor = 30 }
		goto = 3114
		custom_tooltip = vu_pagan.1.tooltip
		3114 = {
			add_base_manpower = 1
			add_local_autonomy = 30
		}
		random_list = {
			50 = {
				3114 = {
					add_local_autonomy = 20
					spawn_rebels = {
						type = norse_pagan_reformed_rebels
						size = 1
					}
				}
				add_adm_power = -10
				add_prestige = -5
			}
			50 = {
				3114 = {
					spawn_rebels = {
						type = norse_pagan_reformed_rebels
						size = 1
					}
				}
				6128 = {
					spawn_rebels = {
						type = norse_pagan_reformed_rebels
						size = 0.5
					}
					add_local_autonomy = 25
				}
				371 = {
					spawn_rebels = {
						type = norse_pagan_reformed_rebels
						size = 0.5
					}
					add_local_autonomy = 25
				}
				add_stability = -1
			}
		}
	}
	
	option = { #Attempt to save the colonies!
		name = vu_pagan.1.c
		highlight = yes
		ai_chance = { factor = 30 }
		1104 = {
			discover_country = ROOT
			cede_province = ROOT
			add_core = ROOT
			change_culture = icelandic
			change_religion = norse_pagan_reformed
			change_trade_goods = whaling	#can also be fur since it was known for good sealing
		}
		add_adm_power = -25
		add_dip_power = -25
		add_treasury = -50
		add_sailors = -250
		add_manpower = -1
	}
}

country_event = {	#Norse communities rise up on Iceland
	id = vu_pagan.2
	title = vu_pagan.2.t
	desc = vu_pagan.2.d
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			owns_core_province = 3114
			owns_core_province = 371
			owns_core_province = 6128
			tag = ICE
		}
		any_owned_province = {
			religion = norse_pagan_reformed
		}
		is_year = 1455
	}
	
	mean_time_to_happen = { #About 5 years
		months = 60
	}
	
	option = { #Agree to their demands
		name = vu_pagan.2.a
		ai_chance = { factor = 45 }
		add_stability = 1
		every_owned_province = {
			limit = { religion = norse_pagan_reformed }
			add_local_autonomy = 40
			add_province_modifier = {
				name = vu_religious_autonomy_modifier
				duration = 14600	#40 years
			}
		}
	}
	
	option = { #Heretics must be purged!
		name = vu_pagan.2.b
		ai_chance = { factor = 45 }
		add_stability = -1
		random_owned_province = {
			limit = { religion = norse_pagan_reformed }
			spawn_rebels = {
				type = norse_pagan_reformed_rebels
				size = 1
			}
		}
		add_country_modifier = {
			name = vu_convert_pagans
			duration = 7300
		}
		add_papal_influence = 15
	}
	
	option = { #Return to the old ways
		name = vu_pagan.2.c
		highlight = yes
		ai_chance = { factor = 10 }
		add_stability = -1
		add_corruption = 1
		capital_scope = { change_religion = norse_pagan_reformed }
		change_religion = norse_pagan_reformed
		add_prestige = 50	#Is to balance -100 prestige from conversion command
		add_country_modifier = {
			name = vu_norse_religion_returns
			duration = 7300
		}
		add_liberty_desire = 25
		custom_tooltip = vu_pagan.2.tooltip
		hidden_effect = {
			every_country = {
				limit = { religion = catholic }
				reverse_add_opinion = {
					who = ROOT
					modifier = turned_norse_pagan
				}
			}
		}
	}
}

country_event = {	#Norse people revolt
	id = vu_pagan.3
	title = vu_pagan.3.t
	desc = vu_pagan.3.d
	picture = RELIGIOUS_TURMOIL_eventPicture
	
	trigger = {
		NOT = { religion = norse_pagan_reformed }
		any_owned_province = {
			religion = norse_pagan_reformed
		}
		is_year = 1455
	}
	
	mean_time_to_happen = { #About 15 years
		months = 180
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					religion = norse_pagan_reformed
				}
				save_event_target_as = vu_norse_revolt
			}
		}
	}
	
	option = { #They must be punished
		name = vu_pagan.3.a
		ai_chance = { 
			factor = 45
			modifier = {
				factor = 50
				ruler_has_personality = cruel_personality
			}
		}
		event_target:vu_norse_revolt = {
			spawn_rebels = {
				type = norse_pagan_reformed_rebels
				size = 1
			}
			add_province_modifier = {
				name = vu_norse_pagan_unhappy
				duration = 1825
				hidden = no
			}
		}
	}
	
	option = { #Bend the knee
		name = vu_pagan.3.b
		ai_chance = {
			factor = 45
			modifier = {
				factor = 0
				ruler_has_personality = cruel_personality
			}
		}
		add_prestige = -10
		add_province_modifier = {
			name = vu_norse_pagan_satisfied
			duration = 1825
			hidden = no
		}
	}
	
	option = { #Negotiate with them
		name = vu_pagan.3.c
		trigger = {
			OR = {
				ruler_has_personality = tolerant_personality
				ruler_has_personality = kind_hearted_personality
				ruler_has_personality = charismatic_negotiator_personality
				ruler_has_personality = just_personality
				ruler_has_personality = lawgiver_personality
			}
		}
		highlight = yes
		ai_chance = { factor = 100 }
		add_adm_power = -10
		add_dip_power = -10
		event_target:vu_norse_revolt = {
			add_province_modifier = {
				name = vu_norse_pagan_stalemate
				duration = 1825
				hidden = no
			}
		}
	}
}
