country_event = {
	id = flavor_gen.3
	title = flavor_gen.3.t
	desc = flavor_gen.3.d
	picture = MERCHANTS_TALKING_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = GEN
		has_government_attribute = is_merchant_republic
		exists = TUR
		TUR = { owns = 151 }			#Constantinople
		151 = { NOT = { months_since_defection = 60 } }	#within 5 years.
		crimea_area = {
			owned_by = GEN
		}
		capital_scope = {
			NOT = { area = crimea_area }
		}
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	option = {
		ai_chance = { factor = 10 }

		name = flavor_gen.3.a
		TUR = {
			country_event = {
				id = flavor_gen.4
				tooltip = flavor_gen.3.4
				days = 1
			}
		}
	}
	option = {
		ai_chance = { factor = 90 }
		name = flavor_gen.3.b
		add_prestige = -10
		every_owned_province = {
			limit = {
				area = crimea_area
			}
			add_province_modifier = {
				name = gen_lost_trade_support
				duration = -1
			}
		}
	}
}
